    public int SingleNumber(int[] nums) 
    {
      Array.Sort(nums);
      int i = 1;
      while (i < nums.Length)
      {
        if (nums[i] - nums[i - 1] != 0)
        {
          return nums[i - 1];
        }
        i += 2;
      }
      return nums[nums.Length - 1];
    }
